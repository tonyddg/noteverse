                                                                # 力学基本定律

                                                                # 自然坐标系 P11

                                                                a⃗=v2Ren⃗+dvdtet⃗ \vec{a} = \frac{v^2}{R} \vec{e_n} + \frac{dv}{dt} \vec{e_t} a=Rv2​en​​+dtdv​et​​

                                                                en⃗\vec{e_n}en​​ 为速度的垂直方向, 指向凹侧, et⃗\vec{e_t}et​​ 为速度方向 et⃗\vec{e_t}et​​ 项的 dvdt\frac{dv}{dt}dtdv​ 为速率(速度的模)的改变量(导数), 不是加速度的模

                                                                # 转动系下的惯性力 P27

                                                                # 离心力

                                                                Fi⃗=12mω2r⃗ \vec{F_i} = \frac{1}{2}m\omega ^2\vec{r} Fi​​=21​mω2r

                                                                r⃗\vec{r}r 为圆心指向物体的矢量, 方向向外, 对于球体则是转轴

                                                                # 科氏力

                                                                Fc⃗=2v⃗×ω⃗ \vec{F_c} = 2\vec{v}\times\vec{\omega} Fc​​=2v×ω

                                                                # 反推力 P35

                                                                F推⃗=udmdt \vec{F_\text{推}} = u\frac{dm}{dt} F推​​=udtdm​

                                                                uuu 为气体相对火箭的速度

                                                                # 动量 P31

                                                                # 动量定理

                                                                注意质量对 dpdpdp 的影响

                                                                F⃗dt=dp⃗ \vec{F}dt=d\vec{p} Fdt=dp​

                                                                # 动量守恒

                                                                ∑p=C \sum p = C ∑p=C

                                                                当合外力为零时使用动量守恒

                                                                # 角动量与力矩 P36

                                                                # 角动量

                                                                L⃗=r⃗×p⃗ \vec{L} = \vec{r} \times \vec{p} L=r×p​

                                                                # 力矩

                                                                M⃗=r⃗×F⃗=dL⃗dt \vec{M} = \vec{r} \times \vec{F} = \frac{d\vec{L}}{dt} M=r×F=dtdL​

                                                                # 角动量定理

                                                                M⃗dt=dL⃗ \vec{M}dt=d\vec{L} Mdt=dL

                                                                # 角动量守恒

                                                                ∑L=C \sum L = C ∑L=C

                                                                当合力通过固定点时使用角动量守恒

                                                                # 保守力做功 P45

                                                                保守力必有两个物体相互作用, 为内力

                                                                # 重力势能

                                                                Ep=mgh E_p = mgh Ep​=mgh

                                                                # 弹性势能

                                                                Ep=12kx2 E_p = \frac{1}{2}kx^2 Ep​=21​kx2

                                                                # 引力势能

                                                                Ep=−GM1M2R E_p = -\frac{GM_1M_2}{R} Ep​=−RGM1​M2​​

                                                                注意引力势能以无穷远为 000 势能点, 有一个负号

                                                                # 功能原理与机械能守恒 P50

                                                                机械能守恒(仅有保守内力做功)

                                                                ∑Ek+Ep=C \sum E_k+E_p=C ∑Ek​+Ep​=C

                                                                功能原理

                                                                A外+A非保守内力=(Ekb+Epb)−(Eka+Epa) A_\text{外}+A_{\text{非保守内力}} = (E_{kb} + E_{pb}) - (E_{ka} + E_{pa}) A外​+A非保守内力​=(Ekb​+Epb​)−(Eka​+Epa​)

                                                                # 转动 P59

                                                                ω=dθdt \omega = \frac{d\theta}{dt} ω=dtdθ​

                                                                α=dωdt \alpha = \frac{d\omega}{dt} α=dtdω​

                                                                v⃗=ω⃗×r⃗ \vec{v} = \vec{\omega} \times \vec{r} v=ω×r

                                                                用于求刚体上某一点的线速度

                                                                # 力矩 P61

                                                                定轴转动的力矩：

                                                                Mz⃗=r⃗×Fv⃗ \vec{M_z} = \vec{r} \times \vec{F_v} Mz​​=r×Fv​​

                                                                其中 r⃗\vec{r}r 为轴指向施力点的方向矢量
                                                                Fv⃗\vec{F_v}Fv​​ 为力垂直于轴平面的分量, 力在轴方向的分量则使刚体沿轴移动

                                                                # 刚体定轴转动定律 P62

                                                                ∣M⃗∣=Jα |\vec{M}| = J\alpha ∣M∣=Jα

                                                                ∣L⃗∣=Jω |\vec{L}| = J\omega ∣L∣=Jω

                                                                此处 L⃗\vec{L}L 为刚体对于轴的角动量

                                                                # 滑轮问题 P67

                                                                在同一条绳子上各点的线加速度与线速度始终相同, 对于滑轮则表现为最外层的线速度与线加速度大小与物体相同. 要使滑轮有线加速度, 两侧的方向相反的力矩必不相同, 体现为滚轮两侧的弹力不同. 体现为：

                                                                RT2−RT1=Jα RT_2 - RT_1 = J\alpha RT2​−RT1​=Jα

                                                                Rα=a R\alpha = a Rα=a
